106 template<typename Ret, typename... Args, class Lamda>
107 struct FuncShadow<Ret (*)(Args...), Lamda>
108 {
109  typedef Ret (*Shadow)(Args...);
110  typedef Ret RetType;
111 
112  static Ret call(Args ...args)
113  {
114  Shadow shadow = &call;
115  long id = (long)shadow;
116  auto iter = stub_wrappers.find(id);
117  assert(stub_wrappers.find(id) != stub_wrappers.end());
118  LamdaWrapper<Lamda> *wrapper = dynamic_cast<LamdaWrapper<Lamda> *>(iter->second);
119  return LamdaCaller<LAMDA_FUNCTION_TYPE>::call(wrapper, args...);
120  }
121 };

156 template<typename Func, class Lamda>
157 typename FuncShadow<Func, Lamda>::Shadow depictShadow(Wrapper **wrapper, Func func, Lamda lamda)
158 {
159  *wrapper = new LamdaWrapper<Lamda>(lamda);
160  typename FuncShadow<Func,Lamda>::Shadow shadow = &FuncShadow<Func,Lamda>::call;
161  long id = (long)shadow;
162  assert(stub_wrappers.find(id) == stub_wrappers.end());
163  stub_wrappers.insert(std::make_pair(id,*wrapper));
164  return shadow;
165 }
