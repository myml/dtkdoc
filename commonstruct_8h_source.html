221 /* 对一些未识别出来的类型进行区分
222 * zip空压缩包：内容检测为"application/octet-stream"，后缀检测为"application/zip"，file命令探测为"application/zip"
223 * 谷歌插件zip：内容检测为"application/octet-stream"，后缀检测为"application/zip"，file命令探测为"application/x-chrome-extension"
224 * 谷歌插件crx：内容检测为"application/octet-stream"，后缀检测为"application/octet-stream"，file命令探测为"application/x-chrome-extension"
225 * zip分卷包：内容检测为"application/octet-stream"，后缀检测为"application/zip"，file命令探测为"application/octet-stream"
226 */
227 struct CustomMimeType
228 {
233  QString name() const
234  {
235  if (m_bUnKnown) {
236  return m_strTypeName;
237  }
238 
239  return m_mimeType.name();
240  }
241 
247  bool inherits(const QString &strMimeType) const
248  {
249  if (m_bUnKnown) {
250  if (strMimeType == m_strTypeName)
251  return true;
252 
253  return false;
254  }
255 
256  return m_mimeType.inherits(strMimeType);
257  }
258  bool m_bUnKnown = false; // 是否自定义扩展类型（内容检测为"application/octet-stream"时使用strTypeName，其余情况使用mimeType）
259  QMimeType m_mimeType;
260  QString m_strTypeName;
261 };
262 Q_DECLARE_METATYPE(CustomMimeType)
