144  inline bool setTo(const DQuickControlColor &color, int colorPropertyIndex) {
145  auto colors = colorPointer(colorPropertyIndex);
146  Q_ASSERT(colors);
147  if (color.data == colors)
148  return color.changed;
149  bool changed = false;
150  const int colorCount = color.isSingleColor ? 1 : ColorFamilyCount;
151  for (int i = 0; i < colorCount; ++i) {
152  if (color.data[i] == colors[i])
153  continue;
154  colors[i] = color.data[i];
155  changed = true;
156  }
157  return changed;
158  }
