323  void setUrls(QList<QUrl> urls)
324  {
325  m_mutex->lock();

327  m_mutex->unlock();
328  };
329  void clearItem()
330  {
331  m_mutex->lock();
332  m_urls.clear();
333  m_mutex->unlock();
334  };
335 
336  void run()
337  {
338  m_mutex->lock();
339  QList<QUrl> urls = m_urls;
340  m_mutex->unlock();
341  foreach (QUrl url, urls) {
342  QFileInfo info(url.path());
343  emit updateItem(m_model->calculatePlayInfo(url, info, false));
344  if (m_stop) break;
345  }
346  }
