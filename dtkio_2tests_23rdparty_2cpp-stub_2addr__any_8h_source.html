206  int get_func_addr(unsigned int ttype, unsigned int stype, std::string& func_name_regex_str, std::map<std::string,void*>& result)
207  {
208  // Create an elfio reader
209  ELFIO::elfio reader;
210  int count = 0;
211  regex_t pathname_regex;
212 
213  if(!m_init)
214  {
215  return -1;
216  }
217 
218  regcomp(&pathname_regex, func_name_regex_str.c_str(), 0);
219  // Load ELF data
220  if(!reader.load(m_fullname.c_str()))
221  {
222  return -1;
223  }
224 
225  ELFIO::Elf_Half sec_num = reader.sections.size();
226  for(int i = 0; i < sec_num; ++i)
227  {
228  ELFIO::section* psec = reader.sections[i];
229  // Check section type
230  if(psec->get_type() == ttype)
231  {
232  const ELFIO::symbol_section_accessor symbols( reader, psec );
233  for ( unsigned int j = 0; j < symbols.get_symbols_num(); ++j )
234  {
235  std::string name;
236  std::string name_mangle;
237  ELFIO::Elf64_Addr value;
238  ELFIO::Elf_Xword size;
239  unsigned char bind;
240  unsigned char type;
241  ELFIO::Elf_Half section_index;
242  unsigned char other;
243 
244  // Read symbol properties
245  symbols.get_symbol( j, name, value, size, bind, type, section_index, other );
246  if(type == STT_FUNC && bind == stype)
247  {
248  bool ret = demangle(name,name_mangle);
249  if(ret == true)
250  {
251  if (0 == regexec(&pathname_regex, name_mangle.c_str(), 0, NULL, 0))
252  {

254  count++;
255  }
256  }

260  {

262  count++;
263  }
264  }
265  }
266  }
267  break;
