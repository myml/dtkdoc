54 { \
55  Q_OBJECT \
56  Q_INTERFACES(QDesignerCustomWidgetInterface) \
57 public: \

60  , m_initialized(false) \
61  { \
62  } \
63  \
64  QString name() const override { \
65  return QStringLiteral("Dtk::Widget::" # UPPER_NAME); \
66  } \
67  \
68  QString group() const override { \
69  return QStringLiteral("Dtk Widgets"); \
70  } \
71  \
72  QString toolTip() const override { \
73  return QStringLiteral("A widgets for dtk gui components."); \
74  } \
75  \

79  \
80  QString includeFile() const override { \
81  return QStringLiteral("" # UPPER_NAME); \

85  return QIcon(QStringLiteral(":/images/" # UPPER_NAME) + ".png"); \
86  } \
87  \
88  bool isContainer() const override { \
89  return false; \
90  } \
91  \
92  UPPER_NAME *createWidget(QWidget *parent) override { \
93  if (IS_SAME(UPPER_NAME, DFrame)) { \
94  return reinterpret_cast<UPPER_NAME *>(new DFrame(parent)); \
95  } \
96  if (IS_SAME(UPPER_NAME, DLabel)) \
97  return reinterpret_cast<UPPER_NAME *>(new DLabel("This is DLabel.", parent)); \
98  if (IS_SAME(UPPER_NAME, DColoredProgressBar)) { \
99  DColoredProgressBar *w = new DColoredProgressBar(parent); \
100  w->setValue(50); \
101  w->setAlignment(Qt::AlignCenter); \
102  return reinterpret_cast<UPPER_NAME *>(w); \
103  } \
104  if (IS_SAME(UPPER_NAME, DFloatingButton)) { \
105  DFloatingButton *w = new DFloatingButton(parent); \
106  w->setIcon(QIcon::fromTheme("lock")); \
107  w->setFixedSize(WIDTH, HEIGHT); \
108  return reinterpret_cast<UPPER_NAME *>(w); \
109  } \
110  if (IS_SAME(UPPER_NAME, DButtonBox)) { \
111  DButtonBox *w = new DButtonBox(parent); \
112  w->resize(WIDTH, HEIGHT); \
113  w->setToolTip("ButtonBox"); \
114  return reinterpret_cast<UPPER_NAME *>(w); \
115  } \
116  if (IS_SAME(UPPER_NAME, DIpv4LineEdit)) { \
117  DIpv4LineEdit *w = new DIpv4LineEdit(parent); \
118  w->resize(WIDTH, HEIGHT); \
119  w->setText("255.255.255.255"); \
120  return reinterpret_cast<UPPER_NAME *>(w); \
121  } \
122  if (IS_SAME(UPPER_NAME, DKeySequenceEdit)) { \
123  DKeySequenceEdit *w = new DKeySequenceEdit(parent); \
124  w->setKeySequence(QKeySequence(Qt::CTRL + Qt::Key_A)); /* 用+号不要用逗号 */ \
125  return reinterpret_cast<UPPER_NAME *>(w); \
126  } \
127  if (IS_SAME(UPPER_NAME, DWaterProgress)) { \
128  DWaterProgress *w = new DWaterProgress(parent); \
129  w->setValue(50); \
130  w->start(); \
131  return reinterpret_cast<UPPER_NAME *>(w); \
132  } \
133  if (IS_SAME(UPPER_NAME, DToolButton)) { \
134  DToolButton *w = new DToolButton(parent); \
135  w->setIcon(QIcon::fromTheme("unlock")); \
136  return reinterpret_cast<UPPER_NAME *>(w); \
137  } \
138  if (IS_SAME(UPPER_NAME, DSpinner)) { \
139  DSpinner *w = new DSpinner(parent); \
140  w->start(); \
141  return reinterpret_cast<UPPER_NAME *>(w); \
142  } \
143  if (IS_SAME(UPPER_NAME, DSuggestButton)) { \
144  DSuggestButton *w = new DSuggestButton("suggest", parent); \
145  return reinterpret_cast<UPPER_NAME *>(w); \
146  } \
147  if (IS_SAME(UPPER_NAME, DWarningButton)) { \
148  DWarningButton *w = new DWarningButton(parent); \
149  w->setText("warning"); \
150  return reinterpret_cast<UPPER_NAME *>(w); \
151  } \
152  \
153  return new UPPER_NAME(parent); \
154  } \
155  \
156  bool isInitialized() const override { return m_initialized;} \
157  void initialize(QDesignerFormEditorInterface */*core*/) override { \
158  if (m_initialized) \
159  return; \
160  m_initialized = true; \
161  } \
162  QString domXml() const override { \
163  QString property; \
164  if (IS_SAME(UPPER_NAME, DColoredProgressBar)) { \
165  property = R"(<property name="value"><number>50</number></property>)"; \
166  } \
167  if (IS_SAME(UPPER_NAME, DProgressBar)) { \
168  property = R"(<property name="value"><number>50</number></property>)"; \
169  } \
170  if (IS_SAME(UPPER_NAME, DLineEdit)) { \
171  property = R"(<property name="text"><number>50</number></property>)"; \
172  } \
173  if (IS_SAME(UPPER_NAME, DIpv4LineEdit)) { \
174  property = R"(<property name="text"><string>255.255.255.255</string></property>)"; \
175  } \
176  if (IS_SAME(UPPER_NAME, DWaterProgress)) { \
177  property = R"(<property name="value"><number>50</number></property>)"; \
178  } \
179  if (IS_SAME(UPPER_NAME, DTextEdit)) { \
180  property = R"(<property name="text"><string>这是一个富文本编辑框</string></property>)"; \
181  } \
182  if (IS_SAME(UPPER_NAME, DSuggestButton)) { \
183  property = R"(<property name="text"><string>suggest</string></property>)"; \
184  } \
185  if (IS_SAME(UPPER_NAME, DWarningButton)) { \
186  property = R"(<property name="text"><string>warning</string></property>)"; \
187  } \
188  return QStringLiteral(" \
189  <ui language=\"c++\" displayname=\"" # UPPER_NAME "\"> \
190  <widget class=\"Dtk::Widget::" # UPPER_NAME "\" name=\"Dtk::Widget::" # UPPER_NAME "\"> \
191  <property name=\"geometry\"> \
192  <rect> \
193  <x>0</x> \
194  <y>0</y> \
195  <width>%1</width> \
196  <height>%2</height> \
197  </rect> \
198  </property> \
199  %3 \
200  </widget> \
201  </ui>").arg(WIDTH).arg(HEIGHT).arg(property); \
202  } \

207 }
